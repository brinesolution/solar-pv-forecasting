import pandas as pd

# 🔹 Path to your broken file
file_path = r"C:\Solar_SUDHA_maam\Solar Irradiation-Power Data\data\Clean15YearChennaiWeather.xls"

# Read file without assuming header
raw_df = pd.read_csv(file_path, header=None)

# If file has 2 broken columns → merge them
if raw_df.shape[1] >= 2:
    raw_df[0] = raw_df[0].astype(str) + raw_df[1].astype(str)

# Now split correctly by comma
df = raw_df[0].str.split(",", expand=True)

# Assign correct column names
df.columns = [
    "Date",
    "Time_Range",
    "Date_ID",
    "Wind",
    "Description",
    "Temperature",
    "Barometric_Pressure",
    "Humidity",
    "Full_Date",
    "Month",
    "Day_Date",
    "Year",
    "Day_Name",
    "Start_Hour",
    "End_Hour"
]

# Strip spaces
df = df.apply(lambda x: x.str.strip() if x.dtype == "object" else x)

# Convert numeric columns
numeric_cols = [
    "Date_ID",
    "Wind",
    "Temperature",
    "Barometric_Pressure",
    "Humidity",
    "Month",
    "Day_Date",
    "Year"
]

for col in numeric_cols:
    df[col] = pd.to_numeric(df[col], errors="coerce")

# Convert date properly
df["Full_Date"] = pd.to_datetime(df["Full_Date"], errors="coerce")

# Save cleaned version
output_path = r"C:\path\to\your\cleaned_weather_data.csv"
df.to_csv(output_path, index=False)

print("File cleaned successfully.")
print("Saved at:", output_path)
print("Shape:", df.shape)
